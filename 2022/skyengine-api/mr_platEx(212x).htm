int32 mr_platEx

(int32 code = 212x,
uint8* input,
int32 input_len,
uint8** output,
int32* output_len,
MR_PLAT_EX_CB* cb
)


Description
获取整首歌的播放时间。
Parameters

Name Description
input 无
input_len 无
output T_MEDIA_TIME数据结构指针
output_len T_MEDIA_TIME数据结构大小（字节数）
cb 无

Return Value

MR_SUCCESS             成功
MR_FAILED                失败
MR_IGNORE                不支持改功能

Remarks

       T_MEDIA_TIME数据结构：
       typedef struct
        {
             int32 pos;
        }T_MEDIA_TIME;

        platEx(212x)其中x代表音频类型的枚举值：
        typedef enum tagT_MEDIA_TYPE
        {
             MEDIA_MIDI = 1,
             MEDIA_WAVE,
             MEDIA_MP3,
             MEDIA_AMR,
             MEDIA_PCM,
             MEDIA_M4A,
             MEDIA_AAC,
             MEDIA_LYRIC,
             MEDIA_UNKNOW
        }T_MEDIA_TYPE;

Example

/*mrp code*/

T_MEDIA_TIME *time = NULL;
int32 len = 0;
mrc_platEx( 2123, NULL, 0, (uint8**)&time, (int32*)&len, NULL );
mrc_printf( "Media total time: %d.", time->pos );
