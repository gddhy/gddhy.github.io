int32 mr_platEx

(int32 code = 1102,
uint8* input,
int32 input_len,
uint8** output,
int32* output_len,
MR_PLAT_EX_CB* cb
)


Description
获得指定一条短信的信息。
Parameters

Name Description
input T_DSM_GET_SMS_INFO_REQ数据结构指针
input_len T_DSM_GET_SMS_INFO_REQ数据结构大小
output T_DSM_MSG_INFO数据结构指针
output_len T_DSM_MSG_INFO数据结构大小
cb 无

Return Value

MR_SUCCESS                    请求成功
MR_FAILED                       请求失败
MR_IGNORE                       不支持该功能

Remarks

短信的数据结构是：T_DSM_MSG_INFO
T_DSM_GET_SMS_INFO_REQ数据结构：
typedef struct
{
    int32 index;           //显示时候索引
    int8 type;              //MR_SMS_INBOX, MR_SMS_OUTBOX
}T_DSM_GET_SMS_INFO_REQ;

T_DSM_MSG_INFO数据结构：
typedef struct
{
    int8 t_stamp [6];              /*t_stamp[0]年t_stamp[1]月t_stamp[2]日 t_stamp[3]时t_stamp[4]分t_stamp[5]秒*/
    int8 num[21];                   /*发送方号码*/ 
    int8 sms_status;               /*Read or Unread */
    int8 sms_storage;             /* MR_SMS_STORAGE_ME , MR_SMS_STORAGE_SIM*/
}T_DSM_MSG_INFO;

Example

/*mrp code*/

T_DSM_GET_SMS_INFO_REQ getInfoReq = { 0 };
T_DSM_MSG_INFO *msgInfo = NULL;
int32 lenmsginfo;

getInfoReq.index = 0;
getInfoReq.type = MR_SMS_OUTBOX;
mrc_platEx( 1102, (uint8* )&getInfoReq, sizeof( T_DSM_GET_SMS_INFO_REQ ), (uint8**)&msgInfo, (int32*)&lenmsginfo, NULL );
