int32 mr_platEx

(int32 code = 4016,
uint8* input,
int32 input_len,
uint8** output,
int32* output_len,
MR_PLAT_EX_CB* cb
)


Description
获得电话本中指定的一个条目的信息。
Parameters

Name Description
input T_MR_PHB_GET_ENTRY_REQ数据结构
input_len T_MR_PHB_GET_ENTRY_REQ数据结构大小
output T_MR_PHB_GET_ENTRY_RSP数据结构指针
output_len T_MR_PHB_GET_ENTRY_RSP数据结构大小
cb 无

Return Value

MR_PHB_OUT_OF_INDEX     (1008)             索引越界
MR_PHB_SUCCESS             (1000)             读取成功
MR_PHB_NOT_READY         (1004)             /*电话本初始化中*/

Remarks

T_MR_PHB_GET_ENTRY_REQ数据结构：
typedef struct
{
    U16 index;          //条目在列表中的索引
    U16  storage;      //条目保存的位置。
}T_MR_PHB_GET_ENTRY_REQ;
注：此处的index 是相对于具体的存储位置而言的，不是在列表中的位置。

T_MR_PHB_GET_ENTRY_RSP数据结构：
typedef struct
{
    T_MR_PHB_ENTRY entry;          //本条目的具体信息
}T_MR_PHB_GET_ENTRY_RSP;
T_MR_PHB_ENTRY数据结构：
typedef struct
{
    S8 name[MR_PHB_MAX_NAME];
    S8 number[MR_PHB_MAX_NUMBER];
    S8 homeNumber[MR_PHB_MAX_NUMBER];
    S8 officeNumber[MR_PHB_MAX_NUMBER];
    S8 faxNumber[MR_PHB_MAX_NUMBER];
    S8 emailAddress[MR_PHB_MAX_NUMBER];
}T_MR_PHB_ENTRY;

电话本接口中对应的参数的定义：
#define MR_PHB_MAX_NAME     (36*2) //暂时定位24个字节,12个汉字.
#define MR_PHB_MAX_NUMBER    (48*2)//暂时定位可输入40个字节.40个数字
#define MR_PHB_MAX_SEARCH_PATTERN  (60*2)

Example

/*mrp code*/

T_MR_PHB_GET_ENTRY_REQ getReq = { 0 };
T_MR_PHB_GET_ENTRY_RSP *getRsp = NULL;
int32 len;

getReq.index = 0;
getReq.storage = 3;
mrc_platEx(4016, (uint8*)&getReq, sizeof ( T_MR_PHB_SEARCH_REQ ), (uint8**)&getRsp, &len, NULL);
